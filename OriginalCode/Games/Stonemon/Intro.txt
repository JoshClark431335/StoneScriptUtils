func Restart()
   fade.index = 15

func Run()
	?fade.state
		>`@centeredX@,@centeredY@,@fade.color@,@scene@
		?fade.index > -10 | sceneState = 22 & fade.index > -20
			fade.chengeIndex(-(totaltime % 2))
		:
			fade.state = false
	:
		?fade.index < 20
			>`@centeredX@,@centeredY@,@fade.color@,@scene@
			fade.chengeIndex(totaltime % 2)
		:
			fade.state = true
			sceneState++
			?sceneState < 23
				scene = intro[sceneState]
			:
				return true
	return false

Restart()
