func RandomVeriance(value, statValue)
	var rngVal = rng
	var startVal = value
	value = value + ((rngVal % ((value/5)+1)) - (value/10))
	value = math.Clamp(value, 0, 999)
	
	Quality = Quality + ((value-startVal)*statValue)
	
	return value

func GenerateCreature(name)
	?name
		GenerateNew(name)
	:
		GenerateNew(rng % 18 + 1)
